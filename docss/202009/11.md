# GitLab CICD 部署 Nginx + Vue

## 方式一

初始化一个 `Vue` 项目作为测试。

> 部署 `nginx`

```
docker pull nginx
```

```
docker run -d -p 8083:80 --name nginx-cicd -v /root/docker/nginx/www:/usr/share/nginx/html nginx
```

> 注册 `runner`

```
docker exec -it gitlab-runner gitlab-runner register -n \
  --url https://gitlab.com/ \
  --registration-token x6LV_YdzNUeEEKpJsaY6 \
  --tag-list=tiny1,tyrande1 \
  --description "hello cicd-vue" \
  --docker-privileged=true \
  --docker-pull-policy="if-not-present" \
  --docker-image "docker:latest" \
  --docker-volumes /var/run/docker.sock:/var/run/docker.sock \
  --docker-volumes /root/docker/nginx/www:/root/docker/nginx/www \
  --executor docker
```

`--docker-volumes /root/docker/nginx/www:/root/docker/nginx/www \` 从容器中映射到 `nginx` 的文件目录下

> `.gitlab-ci.yml`

在项目根目录创建 `.gitlab-ci.yml`

```
stages:
  - build
  - deploy

cache:
  key: ${CI_COMMIT_REF_NAME}
  paths:
    - node_modules/
    - dist

build:
  stage: build
  image: node:latest
  only:
    - master
  script:
    - npm install
    - npm run build
  tags:
    - tiny1

deploy:
  stage: deploy
  only:
    - master
  tags:
    - tiny1
  script:
    - rm -rf /root/docker/nginx/www/*
    - cp -R dist/* /root/docker/nginx/www
  artifacts:
    paths:
      - dist
```

## 方拾二


> 注册 `runner`

```
docker exec -it gitlab-runner gitlab-runner register -n \
  --url https://gitlab.com/ \
  --registration-token x6LV_YdzNUeEEKpJsaY6 \
  --tag-list=tiny1,tyrande1 \
  --description "hello cicd-vue" \
  --docker-privileged=true \
  --docker-pull-policy="if-not-present" \
  --docker-image "docker:latest" \
  --docker-volumes /var/run/docker.sock:/var/run/docker.sock \
  --executor docker
```

> `Dockerfile`

```
FROM nginx:latest

# 设置时区
RUN ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
RUN echo 'Asia/Shanghai' >/etc/timezone

EXPOSE 80
```

> `.gitlab-ci.yml`

```
# 使用docker最新镜像
image: docker:latest

# 定义job执行使用的环境变量
variables:
  DOCKER_DRIVER: overlay2
  # 镜像名称
  IMAGES_NAME: cicd_vue_image
  # 容器名称
  CONTAINER_NAME: CICD_VUE_1.0
  PORT: 8083

cache:
  key: ${CI_COMMIT_REF_NAME}
  paths:
    - node_modules/
    - dist

services:
  - docker:dind

stages:
  - build
  - deploy

build:
  image: node:latest
  stage: build
  tags:
    - tiny1
  script:
    - npm install
    - npm run build
  only:
    - master

deploy:
  services:
    - docker:dind
  stage: deploy
  tags:
    - tiny1
  script:
    # 构件镜像
    - docker build -t $IMAGES_NAME .
    # 删除容器
    - docker rm -f $CONTAINER_NAME || true
    # 运行容器
    - docker run -d --name $CONTAINER_NAME -v /root/docker/nginx/www:/usr/share/nginx/html -p $PORT:80 $IMAGES_NAME
    - cp -R dist/* /root/docker/nginx/www
    # 删除none镜像
    - docker images|grep none|awk '{print $3}'|xargs docker rmi
  allow_failure: true
  only:
    - master
```
